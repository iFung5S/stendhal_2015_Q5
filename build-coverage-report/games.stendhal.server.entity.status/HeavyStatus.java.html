package games.stendhal.server.entity.status;

/**
 * A status ailment that causes the entity to move more slowly
 */
public class HeavyStatus extends Status {
	
	/**
	 * Create the status
	 */
	public HeavyStatus() {
		super("heavy");
	}
	
	/**
	 * @return
	 * 		StatusType
	 */
	@Override
	public StatusType getStatusType() {
		return StatusType.HEAVY;
	}
}
